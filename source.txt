/*the maximum of three numbers*/
int max(int x, int y, int z)
{
	int biggest;
	biggest = x;
	if(y>biggest)  /*comment example*/
		biggest = y;
	if(z>biggest) biggest = z;

	return biggest;
}

{

	int y;
	int z;
	int biggest;

	x = intput();
	y = input();
	z = intput();

	biggest = max(x, y, z);
	output(biggest);
}


int sum(int n)
{
	int result;
	int i;
	
	i = 1;
	result = 0;

	while(i<=n)
	{
		result = result + i;
		i = i + 1;
	}

	return result;
}

void main(void)
{
	int n;
	int s;

	n = intput();
	s = sum(n);
	output(s);
}
